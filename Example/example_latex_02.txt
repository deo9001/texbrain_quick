def fibonacci_sequence(limit: int):
    """Generates Fibonacci numbers up to a specified limit."""
    a, b = 0, 1
    while a < limit:
        yield a
        a, b = b, a + b

# Executing the generator
for number in fibonacci_sequence(100):
    print(f"Validated Sequence Element: {number}")
